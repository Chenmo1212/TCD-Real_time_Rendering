#version 330 core

// Input texture coordinates from vertex shader
in vec2 texCoords;

// Output fragment color
out vec4 color;

// Uniform texture sampler and text color
uniform sampler2D text;
uniform vec3 textColor;

void main() {
    // Sample the texture at the given texture coordinates and only use the red channel
    vec4 sampled = vec4(1.0, 1.0, 1.0, texture(text, texCoords).r);

    // Multiply the sampled texture color with the text color to get the final color
    color = vec4(textColor, 1.0) * sampled;
}
