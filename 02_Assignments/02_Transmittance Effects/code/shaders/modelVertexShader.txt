#version 330
out vec3 normal;
out vec3 position;

in vec3 vertex_position;
in vec3 vertex_normal;

uniform mat4 model;
uniform mat4 view;
uniform mat4 proj;

void main(){
  position = vec3(view * model * vec4(vertex_position, 1.0));
  normal = vec3(view * model * vec4(vertex_normal, 0.0));
  gl_Position = proj * view * model * vec4(vertex_position,1.0);
}
